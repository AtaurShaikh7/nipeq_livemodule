Sub FindCurrDate()

On Error GoTo ErrFindDate

Dim OpBookPath As String
Dim LastLoadStatus As Integer
Dim LastLoadBook As Workbook
Dim WorkDate As Date
Dim WorkDD As String
Dim WorkMM As String
Dim WorkYY As String

LastLoadStatus = 0

'process to find current data date
LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Finding date for which data to load..."
LogBookCounter = LogBookCounter + 1

'find prevload date from database
Dim PrevLoadDayFound As Integer
PrevLoadDayFound = 0
PrevLoadDayFound = FindPrevLoadDay()

If PrevLoadDayFound = 0 Or ErrFlag = 1 Then
    'log already written in database
    LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Prev Load Date not found. Aborting Process...!!!"
    LogBookCounter = LogBookCounter + 1
    Exit Sub
End If

Dim DayPassed As Integer
Dim NSEDataFound As Integer
Dim NSEIndexFilePath As String
Dim NSEIndexFileName As String
Dim DownloadInd As Long
Dim BusinessDayFound As Integer

DayPassed = 1
NSEDataFound = 0
DownloadInd = 1

'fund currdata date from database
BusinessDayFound = 0
BusinessDayFound = FindNextBusinessDay()

If ErrFlag = 1 Then
'    LastLoadBook.Close False
    Exit Sub
End If

If BusinessDayFound = 1 Then
    LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Business Date found for " & Format(CurrDataDate, "DD-MMM-YY") & ". Data upload to be done for this date."
    LogBookCounter = LogBookCounter + 1
Else
    ErrFlag = 1
    'log already written in Find BusinessDay function
End If

If ErrFlag = 0 Then
    
    'find if last load status of database was successful
    Dim DBLastLoad As Integer
    
    DBLastLoad = 0
    
    DBLastLoad = CheckPreviousRunDB
    
    If ErrFlag = 1 Then
        Exit Sub
    End If
    
    If DBLastLoad = 1 Then
        LastLoadStatus = 1
    Else
        'last DB load unsuccessful... abort curr load
        LastLoadStatus = 0
        LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Last Data Load in DataBase unsuccessful... Aborting Process...!!!"
        LogBookCounter = LogBookCounter + 1
        ErrFlag = 1
        DataReady = 0
        Status = "Last Database Data Load for date " & Format(PrevLoadDate, "DD-MMM-YY") & " unsuccessful... Aborting Process...!!!"
        Call WriteLogInDB
        Exit Sub
    End If
    
    If LastLoadStatus = 1 And DBLastLoad = 1 Then
        'prev load was successful
        LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Last load successful... Processing for " & Format(CurrDataDate, "DD-MMM-YY")
        LogBookCounter = LogBookCounter + 1
    End If
End If

Exit Sub

ErrFindDate:
    LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Error in Find Current Data Date"
    LogBookCounter = LogBookCounter + 1
    ErrFlag = 1
    MacroError = 1
    DataReady = 0
    Status = "Macro Error: Error in Find Current Data Date"
    Call WriteLogInDB
End Sub

Sub LoadTradeFile()
On Error GoTo ErrLoadTradeFile

Dim GivenDate As Date

'This is to handle cases of txns coming on non working days
'Typically CBLO txns happen on Sat/Sun
'The process shall loop through all txn files if found from PrevLoadDate + 1 till CurrDataDate
'The txn dates for earlier files shall be set to CurrDataDate

'Start with PrevLoadDate + 1
GivenDate = DateAdd("d", 1, PrevLoadDate)

Do Until GivenDate > CurrDataDate
    'Load for previous dates
    Call LoadGivenDateTradeFile(GivenDate)
    GivenDate = DateAdd("d", 1, GivenDate)
    
Loop

Exit Sub

    ErrFlag = 1
    MacroError = 1
    LogBook.Save
    OutputBook.Save
    DataReady = 0
    Status = "Macro Error: Error in LoadTradeFile"
    Call WriteLogInDB
End Sub

Sub LoadGivenDateTradeFile(CurrTradeFileDate As Date)

Dim TradeFilePath As String
Dim TradeFileName As String
Dim TradeSheetName As String

        OutputBook.Worksheets("TradeData").Activate
        OutputBook.Worksheets("TradeData").Range("A1").Select
        Selection.End(xlDown).Select
        EndRow = ActiveCell.row
    End If

Else

Sub LoadSPBSEIndices()

On Error GoTo ErrLoadSPBSEIndices

Dim PrIndName As String
Dim OPIndName As String
Dim IndexRowCounter As Integer
Dim OpRowCounter As Long
Dim PrOpRowCounter As Long
Dim PrRowCounter As Long
Dim IxRowCounter As Long
Dim IxFileName As String
Dim IxFilePath As String
Dim IxBook As Workbook
Dim IxSheetName As String
Dim PrFileName As String
Dim PrFilePath As String
Dim PrBook As Workbook
Dim PrSheetName As String
Dim PrValue As Double
Dim DownloadInd As Long
Dim PriceDate As Date

Dim PrBookS As Workbook
Dim PrSheetNameS As String

Dim PrBookM As Workbook
Dim PrSheetNameM As String

Dim IndexCounter As Integer
IndexCounter = 2

PrFileName = Format(CurrDataDate, "YYYYMMDD") & ETLMapBook.Worksheets("FILENAMEFORMATS").Range("B10").Value
PrFilePath = ETLMapBook.Worksheets("FILELOCATIONS").Range("B6").Value & PrFileName

'open the Headlines file if it exists
If FileExists(PrFilePath) Then
    'open book
    Workbooks.Open PrFilePath
    Set PrBookM = ActiveWorkbook
    PrSheetNameM = ActiveSheet.Name
Else
    'write error log
    LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Main Index Price file from BSE " & PrFilePath & " not found"
    LogBookCounter = LogBookCounter + 1
    ErrFlag = 1
    DataReady = 0
    Status = "Index Price file from BSE " & PrFilePath & " not found"
    Call WriteLogInDB
    
    Exit Sub
End If

PrFileName = Format(CurrDataDate, "YYYYMMDD") & "_BSE_SECTORS.SDL"
PrFilePath = ETLMapBook.Worksheets("FILELOCATIONS").Range("B6").Value & PrFileName

'open the Sectoral file if it exists
If FileExists(PrFilePath) Then
    'open book
    Workbooks.Open PrFilePath
    Set PrBookS = ActiveWorkbook
    PrSheetNameS = ActiveSheet.Name
Else
    'write error log
    LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Sector Index Price file from BSE " & PrFilePath & " not found"
    LogBookCounter = LogBookCounter + 1
    ErrFlag = 1
    DataReady = 0
    Status = "Index Price file from BSE " & PrFilePath & " not found"
    Call WriteLogInDB
    
    Exit Sub
End If

Do Until ETLMapBook.Worksheets("BSEINDICES").Range("A" & IndexCounter).Value = ""
    
    BSEIndName = ETLMapBook.Worksheets("BSEINDICES").Range("B" & IndexCounter).Value
    PrIndName = ETLMapBook.Worksheets("BSEINDICES").Range("A" & IndexCounter).Value
    OPIndName = ETLMapBook.Worksheets("BSEINDICES").Range("C" & IndexCounter).Value
    
    'open index file
    IxFileName = Format(CurrDataDate, "YYYYMMDD") & "_" & BSEIndName & "_CLS.SDC"
    IxFilePath = ETLMapBook.Worksheets("FILELOCATIONS").Range("B6").Value & IxFileName
    
    'Load constituents if required
    If BSEIndName <> "" And ETLMapBook.Worksheets("BSEINDICES").Range("F" & IndexCounter).Value = "Y" Then
        'Open the file If file doesn't exist write error log and exit
        If FileExists(IxFilePath) And BSEIndName <> "" Then
        
            Workbooks.Open IxFilePath
            Set IxBook = ActiveWorkbook
            IxSheetName = ActiveSheet.Name
        
            LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "BSE index file " & IxFilePath & " found."
            LogBookCounter = LogBookCounter + 1
        
            IndexRowCounter = 2
            OpRowCounter = 2
            PrOpRowCounter = 2
            IxRowCounter = 1
        
            'copy constituent data for this index from index file to ETLDATA
            IxRowCounter = 2
            
            'COPY DATA
            LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Copying Index Constituents for " & OPIndName
            LogBookCounter = LogBookCounter + 1
            
            'find start row in outputbook
            OutputBook.Worksheets("IndexData").Activate
            OutputBook.Worksheets("IndexData").Range("C1").Select
            Selection.End(xlDown).Select
            OpRowCounter = ActiveCell.row
            
            'check if its last row of excel
            If OpRowCounter > 30000 Then
                OpRowCounter = 2
            Else
                OpRowCounter = OpRowCounter + 1
            End If
            
            'COPY DATA
            Do Until UCase(IxBook.Worksheets(IxSheetName).Range("C" & IxRowCounter).Value) = ""
                'copy constituents data
                OutputBook.Worksheets("IndexData").Range("A" & OpRowCounter).Value = OPIndName
                'hardcode EQ for series
                OutputBook.Worksheets("IndexData").Range("B" & OpRowCounter).Value = "EQ"
                'Name
                OutputBook.Worksheets("IndexData").Range("C" & OpRowCounter).Value = IxBook.Worksheets(IxSheetName).Range("E" & IxRowCounter).Value
                'CloseP
                OutputBook.Worksheets("IndexData").Range("D" & OpRowCounter).Value = IxBook.Worksheets(IxSheetName).Range("X" & IxRowCounter).Value
                'Weight
                'Multiply weight by 100 as here it sums up to 1 and system expect 100
                OutputBook.Worksheets("IndexData").Range("E" & OpRowCounter).Value = IxBook.Worksheets(IxSheetName).Range("AH" & IxRowCounter).Value * 100
                
                OutputBook.Worksheets("IndexData").Range("F" & OpRowCounter).Value = Format(CurrDataDate, "DD-MMM-YY")
                 'Put MarketCap, comes in INR, divide by 100,00,000
                OutputBook.Worksheets("IndexData").Range("G" & OpRowCounter).Value = IxBook.Worksheets(IxSheetName).Range("AA" & IxRowCounter).Value / 10000000
                'ISIN
                OutputBook.Worksheets("IndexData").Range("H" & OpRowCounter).Value = IxBook.Worksheets(IxSheetName).Range("I" & IxRowCounter).Value
                'BSE CODE
                OutputBook.Worksheets("IndexData").Range("I" & OpRowCounter).Value = IxBook.Worksheets(IxSheetName).Range("K" & IxRowCounter).Value
                'NSE Symbol not populated for BSE indices
               
                OpRowCounter = OpRowCounter + 1
                IxRowCounter = IxRowCounter + 1
            Loop
        
            'close the index book
            IxBook.Close False
        Else
            'write error log
            LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "BSE index file " & IxFilePath & " not found for " & OPIndName
            LogBookCounter = LogBookCounter + 1
            ErrFlag = 1
            DataReady = 0
            Status = "BSE index file " & IxFilePath & " not found for " & OPIndName
            Call WriteLogInDB
            
            Exit Sub
        End If
    End If
    
    'Load Price if required
    If ETLMapBook.Worksheets("BSEINDICES").Range("D" & IndexCounter).Value = "Y" Then
    
        'set appropriate file to pick data from Headlines/Index
        If ETLMapBook.Worksheets("BSEINDICES").Range("E" & IndexCounter).Value = "Y" Then
            Set PrBook = PrBookS
            PrSheetName = PrSheetNameS
        Else
            Set PrBook = PrBookM
            PrSheetName = PrSheetNameM
        End If
        'copy price data for this index
        LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Getting Price for Index " & OPIndName
        LogBookCounter = LogBookCounter + 1
        
        'find the outputrow in output book for price data
        OutputBook.Worksheets("IndexPrices").Activate
        OutputBook.Worksheets("IndexPrices").Range("C1").Select
        Selection.End(xlDown).Select
        PrOpRowCounter = ActiveCell.row
        
        'check if its last row of excel
        If PrOpRowCounter > 30000 Then
            PrOpRowCounter = 2
        Else
            PrOpRowCounter = PrOpRowCounter + 1
        End If
        
        'copy Price from this file to ETL Data
        OutputBook.Worksheets("IndexPrices").Range("A" & PrOpRowCounter).Value = OPIndName
        OutputBook.Worksheets("IndexPrices").Range("B" & PrOpRowCounter).Value = Format(CurrDataDate, "DD-MMM-YY")
        
        'find the data row for current index and load price
        PrRowCounter = 2
        Do Until (UCase(PrBook.Worksheets(PrSheetName).Range("C" & PrRowCounter).Value) = UCase(PrIndName) Or PrBook.Worksheets(PrSheetName).Range("C" & PrRowCounter).Value = "")
            PrRowCounter = PrRowCounter + 1
        Loop
        
        'Populate Price
        PrValue = 0
        
        If UCase(PrBook.Worksheets(PrSheetName).Range("C" & PrRowCounter).Value) = UCase(PrIndName) Then
            PrValue = PrBook.Worksheets(PrSheetName).Range("H" & PrRowCounter).Value
            OutputBook.Worksheets("IndexPrices").Range("C" & PrOpRowCounter).Value = PrValue
            PrOpRowCounter = PrOpRowCounter + 1
        Else
            'write error log
            LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Price from file " & PrFilePath & " could not be loaded for " & OPIndName
            LogBookCounter = LogBookCounter + 1
            ErrFlag = 1
            DataReady = 0
            Status = "Price from file " & PrFilePath & " could not be loaded for " & OPIndName
            Call WriteLogInDB
            
            PrBook.Close False
            Exit Sub
        End If
                
        LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Data loaded for " & OPIndName
        LogBookCounter = LogBookCounter + 1
    End If
    
    IndexCounter = IndexCounter + 1
Loop

'Close both the price book
PrBookM.Close False
PrBookS.Close False

'write load success message
LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "BSE Indices load successful...!!!"
LogBookCounter = LogBookCounter + 1


OutputBook.Save
LogBook.Save

Exit Sub

ErrLoadSPBSEIndices:
    LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Error in LoadSPBSEIndices"
    LogBookCounter = LogBookCounter + 1
    ErrFlag = 1
    MacroError = 1
    DataReady = 0
    Status = "Macro Error: Error in LoadSPBSEIndices"
