Function CheckPreviousRunDB() As Integer

On Error GoTo ErrCheckPreviousRunDB
    Dim objSession As Object
    Dim objDatabase As Object
    Dim SQL As String
    Dim ORADynaSet As Object
    Dim x As Integer
    Dim y As Integer
    
    CheckPreviousRunDB = 2

    Set objSession = CreateObject("OracleInProcServer.XOraSession")
    Set objDatabase = objSession.OpenDatabase(ORADB, UserID & "/" & Password, 0)
    
    Dim CountFail As Integer
    SQL = "Select Count(*) from Daily_Process_Stats where Status is not null and upper(Process_Name) <> 'DAILY ATTRIB'"
    Set ORADynaSet = objDatabase.DBCreateDynaset(SQL, 0)

    If ORADynaSet.RecordCount > 0 Then
        
        ORADynaSet.MoveFirst

        For y = 0 To ORADynaSet.RecordCount - 1
            For x = 0 To ORADynaSet.Fields.Count - 1
                CountFail = ORADynaSet.Fields(x).Value
            Next
            ORADynaSet.MoveNext
        Next
        If CountFail = 0 Then
            CheckPreviousRunDB = 1
        Else
            CheckPreviousRunDB = 0
        End If
    Else
        CheckPreviousRunDB = 0
    End If

    Set objSession = Nothing
    Set objDatabase = Nothing
    
Exit Function
    
ErrCheckPreviousRunDB:
    Set objSession = Nothing
    Set objDatabase = Nothing
    CheckPreviousRunDB = 0
    LogBook.Worksheets("Sheet1").Range("A" & LogBookCounter).Value = "Error in Check Previous Run in Database"
    LogBookCounter = LogBookCounter + 1
    ErrFlag = 1
    MacroError = 1
    LogBook.Save
    DataReady = 0
    Status = "Macro Error: Error in Check Previous Run in Database"
    
End Function

Function GetEmailIDs() As String

On Error GoTo ErrGetEmailIDs
    Dim objSession As Object
    Dim objDatabase As Object
    Dim SQL As String
    Dim ORADynaSet As Object
    Dim x As Integer
    Dim y As Integer
    
    ORADB = ThisWorkbook.Worksheets("Main").Range("B3").Value
    UserID = ThisWorkbook.Worksheets("Main").Range("B4").Value
    Password = ThisWorkbook.Worksheets("Main").Range("B5").Value
    
    GetEmailIDs = ""

    Set objSession = CreateObject("OracleInProcServer.XOraSession")
    Set objDatabase = objSession.OpenDatabase(ORADB, UserID & "/" & Password, 0)
    
    Dim CountFail As Integer
    SQL = "Select email from alert_emailids"
    Set ORADynaSet = objDatabase.DBCreateDynaset(SQL, 0)

    If ORADynaSet.RecordCount > 0 Then
        
        ORADynaSet.MoveFirst

        For y = 0 To ORADynaSet.RecordCount - 1
            For x = 0 To ORADynaSet.Fields.Count - 1
                GetEmailIDs = GetEmailIDs & ORADynaSet.Fields(x).Value & ";"
            Next
            ORADynaSet.MoveNext
        Next
    Else
        GetEmailIDs = ""
    End If

    Set objSession = Nothing
    Set objDatabase = Nothing
    
Exit Function
    
ErrGetEmailIDs:
    Set objSession = Nothing
    Set objDatabase = Nothing

    LogBookCounter = LogBookCounter + 1
    ErrFlag = 1
    MacroError = 1
    LogBook.Save
    DataReady = 0
    Status = "Macro Error: Error in GetEmailIDs"
    
End Function
